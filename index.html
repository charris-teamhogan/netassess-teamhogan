👤 Active Directory — Users & Groups
▶
Find a user (partial name search) Searches display name, SAM account, UPN — useful when you know part of the name 
# Replace 'Smith' with any partial name, username, or email prefix
$search = 'Smith'

Get-ADUser -Filter {
    (DisplayName -like "*$search*") -or
    (SamAccountName -like "*$search*") -or
    (UserPrincipalName -like "*$search*")
} -Properties DisplayName, Title, Department, Mail, Enabled, LockedOut,
              LastLogonDate, PasswordExpired, PasswordLastSet |
Select-Object Name, SamAccountName, Mail, Title, Department,
              Enabled, LockedOut, LastLogonDate, PasswordExpired |
Format-Table -AutoSize

Get user's group memberships List all groups a specific user belongs to, including nested 
# Replace 'jsmith' with the username (SAM account name)
$user = 'jsmith'

# Direct group memberships
Get-ADUser -Identity $user -Properties MemberOf |
    Select-Object -ExpandProperty MemberOf |
    ForEach-Object { Get-ADGroup $_ } |
    Select-Object Name, GroupScope, GroupCategory |
    Sort-Object Name |
    Format-Table -AutoSize

List group members Show all members of a specific group (e.g. Domain Admins, IT Staff) 
# Replace 'Domain Admins' with the group name
$group = 'Domain Admins'

Get-ADGroupMember -Identity $group -Recursive |
    Where-Object { $_.objectClass -eq 'user' } |
    ForEach-Object {
        Get-ADUser $_.distinguishedName -Properties DisplayName, Mail, Enabled, LastLogonDate
    } |
    Select-Object Name, SamAccountName, DisplayName, Mail, Enabled, LastLogonDate |
    Sort-Object Name |
    Format-Table -AutoSize

Find privileged / admin accounts List members of high-privilege built-in groups 
$privilegedGroups = @(
    'Domain Admins',
    'Enterprise Admins',
    'Schema Admins',
    'Administrators',
    'Account Operators',
    'Backup Operators'
)

foreach ($group in $privilegedGroups) {
    try {
        $members = Get-ADGroupMember -Identity $group -Recursive -ErrorAction Stop |
            Where-Object { $_.objectClass -eq 'user' }
        Write-Host "`n=== $group ($($members.Count) members) ===" -ForegroundColor Yellow
        $members | Select-Object Name, SamAccountName | Format-Table -AutoSize
    } catch {
        Write-Host "Group '$group' not found or no access" -ForegroundColor DarkGray
    }
}

Local network info snapshot IP config, routes, DNS, ARP cache — useful first step on any machine 
# ── IP addresses and adapters ──
Write-Host "=== Network Adapters ===" -ForegroundColor Yellow
Get-NetIPAddress | Where-Object { $_.AddressFamily -eq 'IPv4' -and $_.IPAddress -ne '127.0.0.1' } |
    Select-Object InterfaceAlias, IPAddress, PrefixLength | Format-Table -AutoSize

# ── Default gateway ──
Write-Host "=== Default Gateway ===" -ForegroundColor Yellow
Get-NetRoute -DestinationPrefix '0.0.0.0/0' | Select-Object InterfaceAlias, NextHop | Format-Table -AutoSize

# ── DNS servers ──
Write-Host "=== DNS Servers ===" -ForegroundColor Yellow
Get-DnsClientServerAddress -AddressFamily IPv4 | Where-Object { $_.ServerAddresses } |
    Select-Object InterfaceAlias, ServerAddresses | Format-Table -AutoSize

# ── ARP cache (local subnet hosts) ──
Write-Host "=== ARP Cache ===" -ForegroundColor Yellow
Get-NetNeighbor -AddressFamily IPv4 | Where-Object { $_.State -ne 'Unreachable' } |
    Select-Object InterfaceAlias, IPAddress, LinkLayerAddress, State | Format-Table -AutoSize

# ── Current machine's AD domain ──
Write-Host "=== Domain Info ===" -ForegroundColor Yellow
[System.DirectoryServices.ActiveDirectory.Domain]::GetCurrentDomain() | Select-Object Name, DomainMode, PdcRoleOwner

Test connectivity to a list of hosts Ping sweep / port check against a range — useful for mapping live hosts 
# Edit subnet below — scans .1 to .254
$subnet = '192.168.1'
$results = @()

1..254 | ForEach-Object -Parallel {
    $ip = "$using:subnet.$_"
    $ping = Test-Connection -ComputerName $ip -Count 1 -Quiet -TimeoutSeconds 1
    if ($ping) {
        $dns = try { [System.Net.Dns]::GetHostEntry($ip).HostName } catch { '(no rDNS)' }
        [PSCustomObject]@{ IP = $ip; Hostname = $dns; Alive = $true }
    }
} -ThrottleLimit 50 | Sort-Object { [Version]$_.IP }

# Output results
$results | Format-Table -AutoSize

Export full AD user list to CSV Dump all users with key fields — save the file then upload via OneDrive above 
$outPath = "$env:USERPROFILE\Desktop\ad-users-$(Get-Date -Format 'yyyyMMdd-HHmm').csv"

Get-ADUser -Filter * -Properties DisplayName, Title, Department, Mail,
    Enabled, LockedOut, LastLogonDate, PasswordExpired, PasswordNeverExpires,
    PasswordLastSet, Created, DistinguishedName |
    Select-Object SamAccountName, DisplayName, Title, Department, Mail,
        Enabled, LockedOut, LastLogonDate, PasswordExpired, PasswordNeverExpires,
        PasswordLastSet, Created |
    Export-Csv -Path $outPath -NoTypeInformation -Encoding UTF8

Write-Host "Saved to $outPath" -ForegroundColor Green
